```rust
fn factorial_recursive(i: u64) -> u64 {
    match i {
        0 => 1,
        n => n * factorial(n-1)
    }
}

fn factorial_loop(i: u64) -> u64 {
    let mut acc = 1;
    for num in 2..=i {
        acc *= num;
    }
    acc
}

fn factorial_iter(i: u64) -> u64 {
    (1..=i).product()
}
```
